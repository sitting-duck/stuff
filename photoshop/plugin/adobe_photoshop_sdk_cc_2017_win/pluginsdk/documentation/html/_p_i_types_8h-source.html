00397 /*
00398  * Display Ordering Values.  Passed to certain bottlenecks to tell them what
00399  * the data looks like, and used (on non-Macs) in the pmReserved field of a
00400  * pixmap to tag that data.  For compatibility, the values are designed to
00401  * match up with some Latitude defined values we created for the Unix 3.0.
00402  * - SLB
00403  */
00404  
00405 enum DisplayByteOrder {         /* Data is... */
00406     displayOrderIndexed = 0,    /* 8-bit, or otherwise unspecified. */
00407     displayOrderURGB = 1,       /* 32-bit, high 8 unused, Mac uses this. */
00408     displayOrderUBGR = 2,       /* 32-bit, high 8 unused, Sun, SGI like this. */
00409     displayOrderRGBU = 3,       /* 32-bit, low 8 unused, for completeness. */
00410     displayOrderBGRU = 4,       /* 32-bit, good for little-endian (PC). */
00411     displayOrderRGBPacked = 5,  /* 24-bit, unaligned, currently unsupported!! */
00412     displayOrderBGRPacked = 6   /* 24-bit, unaligned, currently unsupported!! */
